Patch verification for initialization.jl empty reducer crash

The failing function no longer contains sum(_num(v) for v in df[mask, value_col]).
It is implemented as an explicit accumulator loop:

function _excel_sum(df::DataFrame; filters::Dict{Symbol,String}=Dict{Symbol,String}(), value_col::Symbol=:value)
    _hascols(df, [value_col]) || return 0.0
    total = 0.0
    for row in eachrow(df)
        keep = true
        for (col, val) in filters
            if !(col in propertynames(df)) || String(row[col]) != val
                keep = false
                break
            end
        end
        if keep
            total += _num(row[value_col])
        end
    end
    return total
end
